This code chunk takes a single row from metadata and transposes it so we can display a table with it’s entries. We also take the column names from metadata for the labels of that table. Ultimately, this is the R code that we want to write to multiple .qmd files

output <- metadata |>
  filter(vol == "47") |>
  t()


cn <- metadata |>
  colnames()

tibble(cn, output) |>
  gt() |>
  cols_align(align = "left") |>
  tab_style(
    style= cell_text(v_align = "top"),
    locations = cells_body()
  ) |>
  tab_options( table.width = "80%", column_labels.hidden = TRUE)

First, we make sure we can create a single new .qmd file

new_block <- '---
title: "title"
subtitle: "subtitle"
---

## Introduction
This file was generated using R code.

\`\`\`{r}
#| echo: false
#| warning: false

library(tidyverse)
library(gt)

metadata <- read_csv("../contents_raw.csv")
metadata <- metadata |>
  mutate(vol = str_sub(issue, 5, 6))
  
output <- metadata |>
  filter(vol == "47") |>
  t()

cn <- metadata |>
  colnames()

tibble(cn, output) |>
  gt() |>
  cols_align(align = "left") |>
  tab_style(
    style= cell_text(v_align = "top"),
    locations = cells_body()
  ) |>
  tab_options( table.width = "80%", column_labels.hidden = TRUE)
\`\`\`
'

writeLines(new_block, "new_file.qmd")
